public class Business {

    private final Printer printer;

    Business( Printer printer ) {
        this.printer = printer;
    }

    void work() {
        printer.printLn( "Hello");
    }

    /**
     * Do some work with input.
     * @param input to work with
     */

        printer.printLn( "Hello "+input);
    }
    
}
